def create_classifier(input_dim,HIDDEN_LAYER_1=512,HIDDEN_LAYER_2=512,HIDDEN_LAYER_3=128,OUTPUT_LAYER=2):
    classifier=nn.Sequential(
        nn.Linear(input_dim,HIDDEN_LAYER_1),
        nn.ReLU(),
        nn.Dropout(0.2),
        nn.Linear(HIDDEN_LAYER_1,HIDDEN_LAYER_2),
        nn.ReLU(),
        nn.Dropout(0.2),   
        nn.Linear(HIDDEN_LAYER_2,HIDDEN_LAYER_3),
        nn.ReLU(),
        nn.Dropout(0.2),
        nn.Linear(HIDDEN_LAYER_3,OUTPUT_LAYER),
        nn.LogSoftmax(dim=1))
    
    return classifier

def train(model: nn.Module,
          trainloader: torch.utils.data.DataLoader,
          validloader: torch.utils.data.DataLoader,
          optimizer: torch.optim.Optimizer,
          criterion: nn.Module,
          epochs: int,
          print_every: int,
          device: str):
    
    model.to(device)
    
    steps = 0
    running_loss = 0

    f1_metric = BinaryF1Score().to(device)
    precision_metric = BinaryPrecision().to(device)
    recall_metric = BinaryRecall().to(device)
    
    metrics = {
        'train_loss': [],
        'valid_loss': [],
        'valid_accuracy': [],
        'f1_score': [],
        'precision': [],
        'recall': []
    }
    
    all_labels = []
    all_preds = []

    for epoch in range(epochs):
        for images, labels in trainloader:
            model.train() 
            steps += 1
            images, labels = images.to(device), labels.to(device)
            
            optimizer.zero_grad()

            logits = model(images)
            loss = criterion(logits, labels)
            loss.backward()
            optimizer.step()
         
            running_loss += loss.item()
            
            if steps % print_every == 0:
                model.eval()
                accuracy = 0
                valid_loss = 0
                
                with torch.no_grad():
                    for images, labels in validloader:
                        images, labels = images.to(device), labels.to(device)
                    
                        logits = model(images)
                        loss = criterion(logits, labels)
                        valid_loss += loss.item()
                    
                        # Hesaplama adm   
                        ps = torch.exp(logits)
                        top_p, top_class = ps.topk(1, dim=1)
                        equals = top_class == labels.view(*top_class.shape)
                        top_class = top_class.squeeze(1)
                        accuracy += torch.mean(equals.type(torch.FloatTensor)).item()
                        
                        # Metric gncellemesi
                        f1_metric.update(top_class, labels)
                        precision_metric.update(top_class, labels)
                        recall_metric.update(top_class, labels)
                        
                        # 
                        all_labels.extend(labels.cpu().numpy())
                        all_preds.extend(top_class.cpu().numpy())

                    train_loss = running_loss / print_every
                    valid_loss = valid_loss / len(validloader)
                    accuracy = accuracy / len(validloader)
                    
                    f1 = f1_metric.compute()
                    precision = precision_metric.compute()
                    recall = recall_metric.compute()

                    # Kaydetme metrikleri
                    metrics['train_loss'].append(train_loss)
                    metrics['valid_loss'].append(valid_loss)
                    metrics['valid_accuracy'].append(accuracy)
                    metrics['f1_score'].append(f1.item())
                    metrics['precision'].append(precision.item())
                    metrics['recall'].append(recall.item())

                    print(f"Epoch {epoch+1}/{epochs}.. "
                          f"Train loss: {train_loss:.3f}.. "
                          f"Valid loss: {valid_loss:.3f}.. "
                          f"Valid accuracy: {accuracy:.3f}.. "
                          f"F1 Score: {f1:.3f}.. "
                          f"Precision: {precision:.3f}.. "
                          f"Recall: {recall:.3f}")

                    running_loss = 0

    return model, criterion, optimizer.state_dict(), epochs, metrics, all_labels, all_preds

def plot_confusion_matrix(labels, preds,title):
    cm = confusion_matrix(labels, preds)
    plt.figure(figsize=(9,4))
    sns.heatmap(cm, annot=True, fmt="d", cmap="Blues")
    plt.xlabel('Predicted')
    plt.ylabel('True')
    plt.title(title)
    plt.show()

def plot_loss(metrics,title):
    epochs = range(1, len(metrics['train_loss']) + 1)
    
    plt.figure(figsize=(9,4))
    plt.plot(epochs, metrics['train_loss'], label='Train Loss')
    plt.plot(epochs, metrics['valid_loss'], label='Validation Loss')
    
    plt.xlabel('Epochs')
    plt.ylabel('Loss')
    plt.title(title)
    plt.legend()
    plt.show()

def load_checkpoint(checkpoint,dn_model,hidden_layers,output_layer,input_dim):
    
    classifier = nn.Sequential(
    nn.Linear(input_dim, hidden_layers[0]),
    nn.ReLU(),
    nn.Dropout(0.2),
    nn.Linear(hidden_layers[0], hidden_layers[1]),
    nn.ReLU(),
    nn.Dropout(0.2),
    nn.Linear(hidden_layers[1],hidden_layers[2]),
    nn.ReLU(),
    nn.Dropout(0.2),
    nn.Linear(hidden_layers[2],output_layer),
    nn.LogSoftmax(dim=1)
)
    if hasattr(dn_model, 'classifier'):
        dn_model.classifier = classifier
    elif hasattr(dn_model, 'fc'):
        dn_model.fc = classifier
        
    dn_model.load_state_dict(checkpoint["model"])
    optimizer=checkpoint["optimizer_state_dict"]
    criterion = checkpoint["criterion"]
    total_epochs = checkpoint["num_epochs"]
    
    
    return dn_model,optimizer, criterion, total_epochs, hidden_layers, output_layer

def imshow(image, ax=None, title=None):
    """Imshow for Tensor."""
    if ax is None:
        fig, ax = plt.subplots()
    
    # PyTorch tensors assume the color channel is the first dimension
    # but matplotlib assumes is the third dimension
    image = image.transpose((1,2,0))
    
    # Undo preprocessing
    mean = np.array([0.485, 0.456, 0.406])
    std = np.array([0.229, 0.224, 0.225])
    image = std * image + mean
    
    # Image needs to be clipped between 0 and 1 or it looks like noise when displayed
    image = np.clip(image, 0, 1)
    
    ax.imshow(image)
    
    return ax
